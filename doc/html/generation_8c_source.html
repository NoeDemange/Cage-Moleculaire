341 void generateCycle(Shell_t* s) {
342  AtomShl_t* atom;
343  List_t* neighborsNotInCycle;
344  List_t* atomsInCycle = LST_create();
345 
346  // Find the atoms of the shell involved in a cycle.
347  for (int i = 0; i < size(s); i++) {
348  if (flag(atom(s, i)) != NOT_DEF_F) {
349  if (cycle(s, i)) {
350  LST_addElement(atomsInCycle, i);
351  }
352  }
353  }
354 
355  for (int i = 0; forEachElement(atomsInCycle, i); i++) {
356  atom = atom(s, elts(atomsInCycle,i));
357  neighborsNotInCycle = LST_create();
358 
359  // Find the neighbors of the atom not involved in a cycle.
360  for (int j = 0; forEachNeighbor(atom, j); j++) {
361  if (!LST_check(atomsInCycle, neighbor(atom, j)) ||
362  dist(coords(atom), coords(atom(s, neighbor(atom, j)))) > MAXDIS_CYCLE) {
363  LST_addElement(neighborsNotInCycle,neighbor(atom, j));
364  }
365  }
366 
367  int haveEnoughNeighborsInCycle = (SHL_nbNeighborhood(atom) - LST_nbElements(neighborsNotInCycle) >= 2);
368  if (haveEnoughNeighborsInCycle) {
369  // Remove old link between the atom (i) and its neighbors (j).
370  for (int j = 0; forEachElement(neighborsNotInCycle, j); j++) {
371  SHL_removeEdge(s, elts(atomsInCycle,i), elts(neighborsNotInCycle,j));
372  }
373  flag(atom) = CYCLE_F;
374  if (SHL_nbNeighborhood(atom) == 2) {
375  int idNewNeighbor = -1;
376  Point_t newNeighborPoint = addThirdPoint(coords(atom), coords(atom(s, neighbor(atom, 0))),
377  coords(atom(s, neighbor(atom, 1))), SIMPLE_CYCLE);
378  idNewNeighbor = SHL_addAtom(s, newNeighborPoint, -1);
379 
380  for (int j = 0; j < size(s); j++) {
381  if (flag(atom(s, j)) != NOT_DEF_F && dist(newNeighborPoint, coords(atom(s, j))) < MINDIS_CYCLE) {
382  if (LST_check(neighborsNotInCycle, j)) {
383  LST_removeElement(neighborsNotInCycle, j);
384  }
385  if (cycle(s, j)) {
386  //TODO remove this condition unless needed
387  LST_addElement(atomsInCycle, idNewNeighbor);
388  }
389  if (flag(atom(s, j)) != SHELL_F) {
390  SHL_mergeAtom(s, idNewNeighbor, j);
391  }
392  }
393  }
394  if (flag(atom(s, idNewNeighbor)) <= LINKABLE_F) {
395  flag(atom(s, idNewNeighbor)) = LINKABLE_F;
396  }
397  SHL_addEdge(s, elts(atomsInCycle, i), idNewNeighbor);
398  }
399  //SHL_linkBorder(s, elts(atomT,i), nei);
400  }
401  LST_delete(neighborsNotInCycle);
402  }
403  LST_delete(atomsInCycle);
404 }
