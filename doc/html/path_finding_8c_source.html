49 float voxelDist(Point3D a, Point3D b) {
50  float deltaX = fabs((START_GRID + a.x * LENGTH_GRID) - (START_GRID + b.x * LENGTH_GRID));
51  float deltaY = fabs((START_GRID + a.y * LENGTH_GRID) - (START_GRID + b.y * LENGTH_GRID));
52  float deltaZ = fabs((START_GRID + a.z * LENGTH_GRID) - (START_GRID + b.z * LENGTH_GRID));
53  float dmax = (deltaX > deltaY) ? ((deltaX > deltaZ) ? deltaX : deltaZ) : ((deltaY > deltaZ) ? deltaY : deltaZ);
54  float dmin = (deltaX < deltaY) ? ((deltaX < deltaZ) ? deltaX : deltaZ) : ((deltaY < deltaZ) ? deltaY : deltaZ);
55  float dmid = deltaX + deltaY + deltaZ - dmax - dmin;
56  return ((sqrt(3) - sqrt(2)) * dmin + (sqrt(2) - 1) * dmid + dmax);
57 }
