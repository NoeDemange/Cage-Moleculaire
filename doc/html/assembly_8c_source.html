84  // Look for a normal to position the ring.
85  Point_t normal = planNormal(startCycle, start, neighborStart);
86  normal = rotation(vector(startCycle, start), 45, normal);
87

89

91  Point_t newStart, newNeighbor, hydrogen, next;
92 
93  normal = rotation(vector(startCycle, start), 45, normal);
94 
95  Point_t direction = vector(startCycle, start);

97  neighborStart = startCycle;
98

100  continue;
101  }
102 
103  path->patterns[path->size][path->positionNum[path->size]][k++] = newStart;
104 
105  int wasHindered = 0;
106  for (int j = 0; j < 4; j++) {
107 
108  newNeighbor = AX1E2(newStart, neighborStart, normal, SIMPLE_CYCLE);
109  if (j != 2) {
110  hydrogen = AX2E1(newStart, neighborStart, newNeighbor, DIST_ATOM_H);
111  }
112  else {
113  hydrogen = AX2E1(newStart, neighborStart, newNeighbor, DIST_SIMPLE);
114  next = hydrogen;
115  }
116  if (isHindered(processedMoc, substrate, path, newNeighbor) || isHindered(processedMoc, substrate, path, hydrogen)) {
117  wasHindered = 1;
118  break;
119  }
120  if (j != 2) {
121  path->patterns[path->size][path->positionNum[path->size]][k++] = hydrogen;
122  }
123  path->patterns[path->size][path->positionNum[path->size]][k++] = newNeighbor;
124 
125  neighborStart = newStart;
126  newStart = newNeighbor;
127  }
128  if (wasHindered) {
129  continue;
130  }
131  hydrogen = AX2E1(newStart, neighborStart, startCycle, DIST_ATOM_H);
132

134  continue;
135  }
136  path->patterns[path->size][path->positionNum[path->size]][k] = hydrogen;
137  path->patterns[path->size][path->positionNum[path->size]][0] = next;
